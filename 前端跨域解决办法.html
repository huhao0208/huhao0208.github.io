  1. nginx
    反向代理
Nginx conf >> nginx.conf >>
 location  /api {
            rewrite ^/api/(.*)$ /$1 break;
            // 页面地址是a.com，但是要用b.com的cookie
            proxy_cookie_domain b.com a.com;  // 注意别写错位置了 
            proxy_cookie_path / /;
            proxy_pass  https://car.lantin.me/api; # 后端接口 IP:port
        }

2.gulp

var webserver = require("gulp-webserver"); // 下载插件
gulp.task("webserver", function () {
  gulp.src("./").pipe(
    webserver({
      host: "localhost",
      port: 8888,
      livereload: true, // 实时重新加载
      open: "./index.html", // 启动时默认浏览器打开的文件
      directoryListing: {
        enable: true,
        path: "./",
      },
      proxies: [
        {
          source: "/!api",
          target: "http://car.lantin.me", // 代理的域名
        },
      ],
    })
  );
});

3.nodejs

自己简单实现

const target = "https://cn.bing.com"; // 代理地址  localhost/api/
const express = require("express");
const axios = require("axios");
const app = express();
const bodyParser = require("body-parser"); // 解析post请求参数
app.use(bodyParser.urlencoded({ extended: false }));
//设置跨域访问
app.all("*", function (req, res, next) {
  //设置允许跨域的域名，*代表允许任意域名跨域
  res.header("Access-Control-Allow-Origin", "*");
  //允许的header类型
  //   res.header("Access-Control-Allow-Headers", "content-type");
  //跨域允许的请求方式
  res.header("Access-Control-Allow-Methods", "DELETE,PUT,POST,GET,OPTIONS");
  if (req.method.toLowerCase() == "options") res.send(200);
  //让options尝试请求快速结束
  else next();
});
app.use((req, res, next) => {
  const { body, method, headers } = req;
  // 判断 如果是localhost/api 开头的则代理
  let reg = /^\/api\/.*/;
  if (reg.test(req.url)) {
    let targetUlr = "https://cn.bing.com";
    const url = target + req.url.match(/(?<=\/api).*/);
    console.log(url);
    req.url = url;
    axios({
      url,
      method,
      headers: {
        ...headers,
        host: target.match(/(?<=^(https|http)\:\/\/).*/)[0],
      },
      data: Object.keys(body).length ? body : "",
    })
      .then((result) => {
        //   console.log(result.data);
        //   res.status(200).send(JSON.stringify(result.data))
        res.json(result.data);
      })
      .catch((err) => {
        console.log(err);
        res.status(400).json(err);
      });
  } else {
    next();
  }
});
app.listen(80, (_) => {
  console.log("服务启动,端口号80");
});

利用插件 http-proxy-middleware

const express = require("express");
const { createProxyMiddleware } = require("http-proxy-middleware");
const app = express();
//设置跨域访问
app.all("*", function (req, res, next) {
  //设置允许跨域的域名，*代表允许任意域名跨域
  res.header("Access-Control-Allow-Origin", "*");
  //     //允许的header类型
  //  //   res.header("Access-Control-Allow-Headers", "content-type");
  //     //跨域允许的请求方式
  //     res.header("Access-Control-Allow-Methods", "DELETE,PUT,POST,GET,OPTIONS");
  //     if (req.method.toLowerCase() == 'options')
  //         res.send(200);  //让options尝试请求快速结束
  //     else
  //         next();

  next();
});
app.use(
  "/api",
  createProxyMiddleware({
    target: "https://cn.bing.com",
    changeOrigin: true,
    ws: false,
    pathRewrite: {
      "^/api": "",
    },
  })
);
app.listen(80);
