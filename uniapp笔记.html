mixins

分为全局mixins 以及局部mixins

优先级为 先全局再局部 再页面级别

定义

// mixins  定义
export default {

        return {}

使用

// main.js
import xxx from '../../xxx'

    mixins: [xxx], // 也可以引用多个

            // 'scope.writePhotosAlbum'

                                    console.log(result, '权限设置结果');

                                },
                                fail(err) {
                                    console.log(err, 'err');
                                }
                            });
                        }
                    });
                }
