3 网站雏形

// 引入express
const express = require("express");
// 引入qs 处理参数
const querystring = require("querystring");
// 创建网站服务器对象
const app = express();
// 规定求求方式及请求资源定位符
app.use("/index1", (req, res) => {
  // 拦截所有index1 请求
  // req,method 请求方式
});
app.get("/index", (request, response) => {
  response.send("欢迎来到首页");
});
app.post("/form", (req, res) => {
  let params = "";
  // 监听data事件 数据传输时触发
  req.on("data", (p) => {
    params += p;
  });
  // 监听end事件 数据传输完毕触发
  req.on("end", () => {
    res.send(querystring.parse(params));
  });
});
// 监听端口
app.listen(80);
console.log("启动成功");

http 静态资源访问

const http = require("http");
const fs = require("fs");
const Url = require("url");
const path = require("path");
const mime = require("mime");
// 创建网站服务器
const app = http.createServer();
app.on("request", (req, res) => {
  // 获取请求地址
  let url = Url.parse(req.url).pathname;
  url = url === "/" ? "/default.html" : url;
  let type = mime.getType(url); // 获取请求资源类型
  let fullPath = path.join(__dirname, "public", url); // 请求资源绝对地址
  // 读取资源
  fs.readFile(fullPath, (err, result) => {
    // 判断请求结果
    if (err) {
      res.writeHead(404, {
        "content-type": "text/html;charset=utf8",
      });
      res.end(`
            <p>
            资源未找到 404页面
            </p> 
            `);
      return;
    }
    res.writeHead(200, {
      "content-type": type,
    });
    // 响应请求
    res.end(result);
  });
});
app.listen(3000);
console.log("网站服务器启动成功");

const courseSchema = new mongoose.Schema({
  name: String, // 插入数据

// 创建集合并应用规则
const Course = mongoose.model("Course", courseSchema); // 数据库中集合为 courses  第一个参数为集合名称 第二个为集合规则

// 创建文档

集合关联

主要在创建集合规则时关联 id 通过 mongoose.Schema.Types.ObjectId 以及 ref 关联

查询时通过添加 populate(‘要展示的关联字段’)

const mongoose = require("mongoose");

mongoose
  .connect("mongodb://localhost/car")
  .then((res) => {
    console.log("数据库连接成功");
  })
  .catch((err) => {
    console.log(err, "数据库连接失败");
  });
const UserSchema = new mongoose.Schema({
  name: String,
});
const User = mongoose.model("User", UserSchema);
const PostSchema = new mongoose.Schema({
  title: String,
  author: {
    type: mongoose.Schema.Types.ObjectId,
    ref: User,
  },
});
const Post = mongoose.model("Post", PostSchema);
// // 创建用户
User.create({
  name: "文章1",
});
// 创建文章
Post.create({
  title: "test文章标题",
  author: "5f0c0f881c7b56484019ac54",
});
// 查询文章 并关联作者
Post.find()
  .populate("author")
  .then((res) => {
    console.log(res);
  });
